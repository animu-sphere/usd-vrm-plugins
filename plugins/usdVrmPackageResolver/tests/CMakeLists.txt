# Plain-CMake (non-OpenStrata) test wiring. The authoritative verification path
# is `ost plugin test plugins/usdVrmPackageResolver` (the L0-L6 pyramid); these
# CTest targets keep the bare-CMake flow end-to-end testable and add the
# boundary check the pyramid does not own.

# Resolve the OpenUSD install root from the pxr CMake package location (same
# approach as plugins/usdVrm/tests).
set(_usd_root "")
if(pxr_DIR AND EXISTS "${pxr_DIR}/bin")
    set(_usd_root "${pxr_DIR}")
elseif(pxr_DIR)
    get_filename_component(_usd_root "${pxr_DIR}/../../.." ABSOLUTE)
endif()

find_package(Python3 COMPONENTS Interpreter QUIET)
if(NOT Python3_Interpreter_FOUND)
    message(STATUS "usdVrmPackageResolver: Python3 not found; skipping tests")
    return()
endif()

# WORKSPACE.md §2 link check: the built resolver must import vrmContainer and
# must not import the file format or the schema library.
add_test(NAME usdvrmresolver_boundaries
    COMMAND "${Python3_EXECUTABLE}"
        "${CMAKE_CURRENT_SOURCE_DIR}/check_boundaries.py"
        "${CMAKE_CURRENT_SOURCE_DIR}/.."
        "$<TARGET_FILE:${PLUGIN_NAME}>")

# Functional + security suite, driven through the real Ar surface with ONLY
# this bundle registered (no file format, no schema).
set(_bundle_dir "${CMAKE_CURRENT_SOURCE_DIR}/..")

# Build-tree plugInfo with an absolute LibraryPath, so the test does not depend
# on the source-tree plugInfo generated by the ost flow. The staged library
# location is fixed for every configuration by openstrata_configure_plugin
# (bundle lib/, "lib" prefix). CMake paths use forward slashes everywhere, so
# the JSON stays valid.
set(USDVRM_RESOLVER_TEST_LIBRARYPATH
    "${_bundle_dir}/lib/lib${PLUGIN_NAME}${CMAKE_SHARED_LIBRARY_SUFFIX}")
set(_plug_resources "${CMAKE_CURRENT_BINARY_DIR}/resources/usdVrmPackageResolver")
configure_file(
    "${CMAKE_CURRENT_SOURCE_DIR}/plugInfo.in.json"
    "${_plug_resources}/plugInfo.json"
    @ONLY)

add_test(
    NAME usdvrmresolver_python
    COMMAND ${Python3_EXECUTABLE}
            "${CMAKE_CURRENT_SOURCE_DIR}/test_resolver.py")

# vrmContainer's DLL/SO location: in-tree target dir in the composed root
# build, installed workspace-prefix bin/ in the ost standalone flow.
set_tests_properties(usdvrmresolver_python PROPERTIES
    ENVIRONMENT_MODIFICATION
        "PATH=path_list_prepend:${_usd_root}/bin;PATH=path_list_prepend:${_usd_root}/lib;PATH=path_list_prepend:$<TARGET_FILE_DIR:${PLUGIN_NAME}>;PATH=path_list_prepend:$<TARGET_FILE_DIR:vrmContainer::vrmContainer>;PYTHONPATH=path_list_prepend:${_usd_root}/lib/python;PYTHONPATH=path_list_prepend:${_usd_root}/lib/site-packages;PXR_PLUGINPATH_NAME=path_list_prepend:${_plug_resources}")
